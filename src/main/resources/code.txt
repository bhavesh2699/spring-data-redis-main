import React, { useState, useEffect } from 'react';
import axios from 'axios';

const JSONParserExample = () => {
  const [jsonData, setJsonData] = useState(null);
  const [owner, setOwner] = useState('');

  useEffect(() => {
    const fetchData = async () => {
      try {
        const response = await axios.get('https://your-api-endpoint.com/data'); // Replace with your API endpoint
        const data = response.data;
        setJsonData(data);

        // Extract colIdx values for "code" and "owner"
        const { columns: { inputs } } = data;
        const codeColIdx = inputs.find(input => input.colName === 'code').colIdx;
        const ownerColIdx = inputs.find(input => input.colName === 'owner').colIdx;

        // Search for "DD_Test" and retrieve owner details
        const { data: { rows } } = data;
        const matchingRow = rows.find(row => row.cols.some(col => col.colIdx === codeColIdx && col.cellData === 'DD Test'));
        if (matchingRow) {
          setOwner(matchingRow.cols.find(col => col.colIdx === ownerColIdx).cellData);
        }
      } catch (error) {
        console.error('Error fetching data:', error);
      }
    };

    fetchData();
  }, []);

  return (
    <div>
      {jsonData ? (
        <p>Owner: {owner}</p>
      ) : (
        <p>Loading data...</p>
      )}
    </div>
  );
};

export default JSONParserExample;
